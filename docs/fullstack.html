Webpack - быстрая сборка

Быстрая сборка

  • npm init -y

  • npm i -D webpack webpack-cli webpack-dev-server html-webpack-plugin html-loader style-loader mini-css-extract-plugin sass-loader sass postcss-loader postcss postcss-preset-env babel-loader @babel/core @babel/preset-env image-webpack-loader css-loader

  • npm install --save @babel/polyfill normalize.css

  • В SCSS-файле прописать:

    @use "normalize.css";

  • В JS-файле прописать:

    // import '../index.html';
import '../scss/styles.scss';

  • Создать файл .browserslistrc прописать:

    last 2 versions
not dead
> 0.5%

  • В package.json прописать:

    "scripts": {
  "start": "webpack serve",
  "build-dev": "webpack",
  "build-prod": "webpack --node-env=production",
  "clear": "rd /s /q docs"
},

  • webpack.config.js одностроничного сайта:

    const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssExtractPlugin = require("mini-css-extract-plugin");
const loader = require('sass-loader');
const mode = process.env.NODE_ENV || 'development';
const devMode = mode === 'development';
const target = devMode ? 'web' : 'browserslist';
const devtool = devMode ? 'source-map' : undefined;
function getRandomFourDigitNumber() {
  return Math.floor(Math.random() * 9000) + 1000;
}
module.exports = {
  mode,
  target,
  devtool,
  devServer: {
    port: getRandomFourDigitNumber(),
    open: true,
    hot: true,
  },
  entry: ['@babel/polyfill', path.resolve(__dirname, 'src', 'js', 'scripts.js')],
  output: {
    path: path.resolve(__dirname, 'docs'),
    clean: true,
    filename: 'scripts.[contenthash].js',
    assetModuleFilename: 'images/[name][ext]'
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: path.resolve(__dirname, 'src', 'index.html'),
    }),
    new MiniCssExtractPlugin({
      filename: 'styles.[contenthash].css',
    })
  ],
  module: {
    rules: [
      {
        test: /\.html$/i,
        loader: "html-loader",
      },
      {
        test: /\.(c|sa|sc)ss$/i,
        use: [
          devMode ? "style-loader" : MiniCssExtractPlugin.loader,
          "css-loader",
          {
            loader: "postcss-loader",
            options: {
              postcssOptions: {
                plugins: [require("postcss-preset-env")]
              }
            }
          },
          "sass-loader"],
      },
      {
        test: /\.(woff(2)?|eot|ttf|otf|svg|)$/i,
        type: 'asset/resource',
        generator: {
          filename: 'fonts/[name][ext]'
        }
      },
      {
        test: /\.(png|svg|jpg|jpeg|gif|webp)$/i,
        use: [
          {
            loader: 'image-webpack-loader',
            options: {
              mozjpeg: {
                progressive: true,
              },
              optipng: {
                enabled: false,
              },
              pngquant: {
                quality: [0.65, 0.90],
                speed: 4
              },
              gifsicle: {
                interlaced: false,
              },
              webp: {
                quality: 75
              }
            }
          }
        ],
        type: "asset/resource",
      },
      {
        test: /\.(?:js|mjs|cjs)$/,
        exclude: /(node_modules|bower_components)/,
        use: {
          loader: 'babel-loader',
          options: {
            targets: "defaults",
            presets: [
              ['@babel/preset-env']
            ]
          }
        }
      }
    ],
  },
}

const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssExtractPlugin = require("mini-css-extract-plugin");

const mode = process.env.NODE_ENV || 'development';
const devMode = mode === 'development';
const target = devMode ? 'web' : 'browserslist';
const devtool = devMode ? 'source-map' : undefined;

function getRandomFourDigitNumber() {
  return Math.floor(Math.random() * 9000) + 1000;
}

module.exports = {
  mode,
  target,
  devtool,
  devServer: {
