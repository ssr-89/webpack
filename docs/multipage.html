Webpack и Pug

Проект на Webpack с установкой шаблонизатора Pug.

Настройка Webpack и Pug

  • npm install --save-dev webpack webpack-cli webpack-dev-server

  • Пример структуры сайта

    src
--layouts
----main.pug
--pages
----index.pug
----about.pug
--components
----header.pug
----footer.pug

  • Пример структуры main.pug

    doctype html
html(lang="ru")
head
meta(charset="UTF-8")
title Webpack и Pug
body
  div#wrapper.wrapper
    include ../components/header.pug
    main#main.main
    block content
    include ../components/footer.pug

  • Пример структуры header.pug

    header#header.header
  div.header__container
    nav.header__nav
      ul.header__menu.header-menu.list
        li.header-menu__item
          a(href='./index.html').header-menu__link.link

        li.header-menu__item
          a(href='./about.html').header-menu__link.link
            span.header-menu__link-text О нас

  • Пример структуры footer.pug

    footer#footer.footer
  div.footer__container
    h2 This is FOOTER

  • Полный webpack.config.js:

    const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssExtractPlugin = require("mini-css-extract-plugin");
const loader = require('sass-loader');

const mode = process.env.NODE_ENV || 'development';
const devMode = mode === 'development';
const target = devMode ? 'web' : 'browserslist';
const devtool = devMode ? 'source-map' : undefined;

function getRandomFourDigitNumber() {
  return Math.floor(Math.random() * 9000) + 1000;
}

module.exports = {
  mode,
  target,
  devtool,
  devServer: {
    port: getRandomFourDigitNumber(),
    open: true,
    hot: true,
  },
  entry: ['@babel/polyfill', path.resolve(__dirname, 'src', 'js', 'scripts.js')],
  output: {
    path: path.resolve(__dirname, 'docs'),
    clean: true,
    filename: 'scripts.[contenthash].js',
    assetModuleFilename: 'images/[name][ext]'
  },
  plugins: [
    new HtmlWebpackPlugin({
      filename: 'index.html',
      template: path.resolve(__dirname, 'src', 'pages', 'index.pug'),
      inject: 'body',
      templateParameters: {
        title: "Webpack"
      }
    }),
    new HtmlWebpackPlugin({
      filename: 'multipage.html',
      template: path.resolve(__dirname, 'src', 'pages', 'multipage.pug'),
      inject: 'body',
      templateParameters: {
        title: "Webpack и Pug"
      }
    }),
    new MiniCssExtractPlugin({
      filename: 'styles.[contenthash].css',
    })
  ],
  module: {
    rules: [
      {
        test: /.pug$/,
        use: ['pug-loader'],
      },
      {
        test: /\.html$/i,
        loader: "html-loader",
      },
      {
        test: /\.(c|sa|sc)ss$/i,
        use: [
          devMode ? "style-loader" : MiniCssExtractPlugin.loader,
          "css-loader",
          {
            loader: "postcss-loader",
            options: {
              postcssOptions: {
                plugins: [require("postcss-preset-env")]
              }
            }
          },
          "sass-loader"],
      },
      {
        test: /\.(woff(2)?|eot|ttf|otf|svg|)$/i,
        type: 'asset/resource',
        generator: {
          filename: 'fonts/[name][ext]'
        }
      },
      {
        test: /\.(png|svg|jpg|jpeg|gif|webp)$/i,
        use: [
          {
            loader: 'image-webpack-loader',
            options: {
              mozjpeg: {
                progressive: true,
              },
              optipng: {
                enabled: false,
              },
              pngquant: {
                quality: [0.65, 0.90],
                speed: 4
              },
              gifsicle: {
                interlaced: false,
              },
              webp: {
                quality: 75
              }
            }
          }
        ],
        type: "asset/resource",
      },
      {
        test: /\.(?:js|mjs|cjs)$/,
        exclude: /(node_modules|bower_components)/,
        use: {
          loader: 'babel-loader',
          options: {
            targets: "defaults",
            presets: [
              ['@babel/preset-env']
            ]
          }
        }
      }
    ],
  },
}
